spanner:var_type

ACTION: walk(start:location,end:location,m:man)
TYPE: memory
PRECONDITIONS:
( atm(m,start) = 1 )
( link(start,end) = 1 )
EFFECTS:
( atm(m,start) = 0 )
( atm(m,end) = 1 )

ACTION: pickup_spanner(l:location,s:spanner,m:man)
TYPE: memory
PRECONDITIONS:
( atm(m,l) = 1 )
( ats(s,l) = 1 )
EFFECTS:
( ats(s,l) = 0 )
( carrying(m,s) = 1 )

ACTION: tighten_nut(l:location,s:spanner,m:man,n:nut)
TYPE: memory
PRECONDITIONS:
( atm(m,l) = 1 )
( atn(n,l) = 1 )
( carrying(m,s) = 1 )
( useable(s) = 1 )
( loose(n) = 1 )
EFFECTS:
( loose(n) = 0 )
( useable(s) = 0 )
( tightened(n) = 1 )
